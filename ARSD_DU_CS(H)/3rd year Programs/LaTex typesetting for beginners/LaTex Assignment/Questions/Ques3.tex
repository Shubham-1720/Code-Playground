\section*{Question:3 Write LaTex command to produce.}
\textbf{(i) Write the command to produce}
\[
\delta_{ij} = 
\begin{cases} 
0 & \text{if } i \neq j, \\
1 & \text{if } i = j.
\end{cases}
\]

\textbf{(ii) Write the command to produce}
\[
x^2 \frac{d^2y}{dx^2} + x \frac{dy}{dx} + (x^2 - \nu^2)y = 0
\]
The solution of which is:
\[
y = AJ_\nu(x) + BY_\nu(x),
\]
\[
= AJ_\nu(x) + BY_{-\nu}(x),
\]
\vspace{0.5cm}
if $\nu$ is non-integer.\\

\textbf{Answer:-}
\section*{Command to produce}
\subsection*{(i)}
\begin{verbatim}
\[
\delta_{ij} = 
\begin{cases} 
0 & \text{if } i \neq j, \\
1 & \text{if } i = j.
\end{cases}
\]

\end{verbatim}
\vspace{0.5cm}
\subsection*{(ii)}
\begin{verbatim}
\[
x^2 \frac{d^2y}{dx^2} + x \frac{dy}{dx} + (x^2 - \nu^2)y = 0
\]
The solution is:
\[
y = AJ_\nu(x) + BY_\nu(x),
\]
\[
= AJ_\nu(x) + BY_{-\nu}(x),
\]  
\end{verbatim}
